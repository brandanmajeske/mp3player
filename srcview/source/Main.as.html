package
{
    import flash.display.Sprite;
    import com.brandanmajeske.app.MusicPlayer;
    
    [SWF("800", height="400", backgroundColor="#000000", frameRate="60")]
    
    public class Main extends Sprite
    {
        public function Main()
        {

            //add Audio app to the stage
            var mp:MusicPlayer = new MusicPlayer();
            mp.x = (stage.stageWidth-mp.width)/2;
            mp.y = (stage.stageHeight-mp.height)/2;
            this.addChild(mp);
        }
    }
}
